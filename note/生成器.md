### 生成器
**1.介绍**
生成器其实是一个特殊函数，用于异步编程
【⚠️】生成器函数声明方式： function 和函数名中间有个*
【⚠️】生成器函数调用，单纯调用方法不行，要调用方法后加next()
【⚠️】生成器函数中可以有yield语句，yield算作是函数代码的分隔符，一个yield产生2块代码，2个分隔符产生3块代码
```
function * gen(){
    console.log('111')
    yield '这是第一句'
    yield '这是第二句'
}
let iterator = gen()
iterator.next()     
console.log(iterator.next())
// 111 
```
```
function * gen(){
    console.log('111')
    yield '这是第一句'
    yield '这是第二句'
}
let iterator = gen()
iterator.next() 
iterator.next()  
// 111
// 222 
```
```
function * gen(){
    console.log('111')
    yield '这是第一句'
    console.log('222')
    yield '这是第二句'
    console.log('333')
}
let iterator = gen()
iterator.next() 
iterator.next()    
iterator.next()   
// 111
// 222
// 333
```
以上是验证生成器函数是通过next()方法向下执行，也看出了是个迭代器对象，故可以用 for...of
```
 function * gen(){
        console.log('111')
        yield '这是第一句'
        console.log('222')
        yield '这是第二句'
        console.log('333')
    }
    for(let v of gen()){
        console.log(v)
    }
    // 111
    // 这是第一句
    // 222
    // 这是第二句
    ///333
```

**2.生成器函数的传参**
两种传参方式，next()方法传参将作为上一个yield语句的返回结果，可以实现异步编程参数传递的效果。
```
     function * gen(arg){
        console.log('arg:' + arg)
        let one = yield 111;
        console.log('one:' + one)
        let two = yield 222;
        console.log('two:' + two)
    }
    // 执行获取迭代器对象
    let iterator = gen('aaa')
    iterator.next()  // arg:aaa
```
```
    function * gen(arg){
        console.log('arg:' + arg)
        let one = yield 111;
        console.log('one:' + one)
        let two = yield 222;
        console.log('two:' + two)
    }
    let iterator = gen('aaa')
    // next()方法可以传入实参
    iterator.next('bbb')  // arg:aaa
```
```
    function * gen(arg){
        console.log('arg:' + arg)
        let one = yield 111;
        console.log('one:' + one)
        let two = yield 222;
        console.log('two:' + two)
    }
    let iterator = gen('aaa')
    iterator.next()       // arg:aaa
    iterator.next('')     // one:
    iterator.next('bbb')  // two:bbb
    iterator.next('ccc')  // 无打印信息  
```

**3.生成器函数实例1**
要实现 1秒钟后显示111，再2秒钟后显示222，再三秒钟后显示333
```
// 回调地狱写法
setTimeout(() => {
        console.log('111')
        setTimeout(() => {
            console.log('222')
            setTimeout(() => {
                console.log('333')
            },1000)
        },1000)
    },1000)
```
```
// 生成器函数写法（并没有实现需求）
    function one(){
        setTimeout(() => {
            console.log('111')
        },1000)
    }
    function two(){
        setTimeout(() => {
            console.log('222')
        },2000)
    }
    function three(){
        setTimeout(() => {
            console.log('333')
        },3000)
    }
    function * gen(){
        yield one()
        yield two()
        yield three()
    }
    let iterator = gen()
    /*
    * 这种写法会每隔一秒出现
    * 111
    * 222
    * 333
    * */
    iterator.next()
    iterator.next()
    iterator.next()
```

**4.生成器函数实例2**
生成器函数在异步任务中的表现，需要第一个方法中的数据带到第二个方法中，第二个方法中的数据带到第三个方法中
```
    function one(){
        setTimeout(() => {
            console.log('初始数据')
           let data = '初始数据'
           iterator.next(data)
        },1000)
    }
    function two(){
        setTimeout(() => {
            console.log('生成订单')
            let data = '生成订单'
            iterator.next(data)
        },1000)
    }
    function three(){
        setTimeout(() => {
            console.log('打包商品')
            let data = '打包商品'
            iterator.next(data)
        },1000)
    }
    function * gen(){
       yield one()
       yield two()
       yield three()
    }
    let iterator = gen()
    iterator.next()
    /*
    * 每间隔一秒打印出
    *  初始数据
    *  生成订单
    *  打包商品
    * */
```
